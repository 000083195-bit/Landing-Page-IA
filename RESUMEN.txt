🚀 INICIO RÁPIDO
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

LOCAL:
1. Ejecutar database.sql en MySQL
2. Copiar .env.example a .env
3. Editar .env con credenciales de MySQL local
4. Iniciar servidor PHP:
   $ php -S localhost:8000
5. Abrir http://localhost:8000/login.php

RAILWAY:
1. Conectar repo a Railway
2. Agregar servicio MySQL
3. Copiar credenciales a variables de entorno
4. Ejecutar database.sql en BD de Railway
5. Deploy automático

CREDENCIALES DE PRUEBA:
- Admin: admin@pesado.com / password123
- User: user@pesado.com / password123

💾 ESTRUCTURA BASE DE DATOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

TABLA: usuarios
  ├─ id (INT, PRIMARY KEY)
  ├─ nombre (VARCHAR)
  ├─ email (VARCHAR, UNIQUE)
  ├─ contraseña (VARCHAR, hasheada)
  ├─ rol (ENUM: admin, user)
  ├─ estado (ENUM: activo, inactivo)
  ├─ fecha_creacion (TIMESTAMP)
  └─ fecha_actualizacion (TIMESTAMP)

TABLA: ventas
  ├─ id (INT, PRIMARY KEY)
  ├─ fecha (DATE)
  ├─ producto (VARCHAR)
  ├─ cantidad (INT)
  ├─ precio_unitario (DECIMAL)
  ├─ total (DECIMAL)
  ├─ cliente_email (VARCHAR)
  ├─ metodo_pago (VARCHAR)
  ├─ estado (ENUM: completada, pendiente, cancelada)
  └─ fecha_creacion (TIMESTAMP)

TABLA: productos
  ├─ id (INT, PRIMARY KEY)
  ├─ nombre (VARCHAR)
  ├─ descripcion (TEXT)
  ├─ categoria_id (INT, FOREIGN KEY)
  ├─ precio (DECIMAL)
  ├─ stock (INT)
  └─ estado (ENUM: activo, inactivo)

TABLA: categorias
  ├─ id (INT, PRIMARY KEY)
  ├─ nombre (VARCHAR)
  ├─ descripcion (TEXT)
  └─ estado (ENUM: activo, inactivo)

TABLA: sesiones
  ├─ id (INT, PRIMARY KEY)
  ├─ usuario_id (INT, FOREIGN KEY)
  ├─ token (VARCHAR, UNIQUE)
  ├─ ip_address (VARCHAR)
  ├─ user_agent (VARCHAR)
  ├─ fecha_creacion (TIMESTAMP)
  └─ fecha_expiracion (DATETIME)
